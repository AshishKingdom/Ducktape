69 Component *Assign(const std::string &name)
70 {
71 AssignFunc assignFunc = (AssignFunc)GetProcAddress(scene->gameModule, ("Assign" + name).c_str());

73 if (!assignFunc)

75 std::cout << "Failed to get Assign" << name << "(Entity entity) function from Game Module." << std::endl;
76 return nullptr;
