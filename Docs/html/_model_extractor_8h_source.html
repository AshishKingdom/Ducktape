91 Mesh ProcessMesh(aiMesh *mesh, const aiScene *scene)
92 {
93 // data to fill
94 std::vector<Vertex> vertices;
95 std::vector<unsigned int> indices;
96 std::vector<Texture> textures;
97
98 // walk through each of the mesh's vertices
99 for (unsigned int i = 0; i < mesh->mNumVertices; i++)
100 {
101 Vertex vertex;
102 glm::vec3 vector; // we declare a placeholder vector since assimp uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
103 // positions
104 vector.x = mesh->mVertices[i].x;
105 vector.y = mesh->mVertices[i].y;
106 vector.z = mesh->mVertices[i].z;
107 vertex.position = vector;
108 // normals
109 if (mesh->HasNormals())
110 {
111 vector.x = mesh->mNormals[i].x;
112 vector.y = mesh->mNormals[i].y;
113 vector.z = mesh->mNormals[i].z;
114 vertex.normal = vector;
115 }
116 // texture coordinates
117 if (mesh->mTextureCoords[0]) // does the mesh contain texture coordinates?
118 {
119 glm::vec2 vec;
120 // a vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
121 // use models where a vertex can have multiple texture coordinates so we always take the first set (0).
122 vec.x = mesh->mTextureCoords[0][i].x;
123 vec.y = mesh->mTextureCoords[0][i].y;
124 vertex.texCoords = vec;
125 // tangent
126 vector.x = mesh->mTangents[i].x;
127 vector.y = mesh->mTangents[i].y;
128 vector.z = mesh->mTangents[i].z;
129 vertex.tangent = vector;
130 // bitangent
131 vector.x = mesh->mBitangents[i].x;
132 vector.y = mesh->mBitangents[i].y;
133 vector.z = mesh->mBitangents[i].z;
134 vertex.bitangent = vector;
135 }
136 else
137 vertex.texCoords = glm::vec2(0.0f, 0.0f);
138
139 vertices.push_back(vertex);
140 }
141 // now wak through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
142 for (unsigned int i = 0; i < mesh->mNumFaces; i++)
143 {
144 aiFace face = mesh->mFaces[i];
145 // retrieve all indices of the face and store them in the indices vector
146 for (unsigned int j = 0; j < face.mNumIndices; j++)
147 indices.push_back(face.mIndices[j]);
148 }
149 // process materials
150 aiMaterial *material = scene->mMaterials[mesh->mMaterialIndex];
151 // we assume a convention for sampler names in the shaders. Each diffuse texture should be named
152 // as 'texture_diffuseN' where N is a sequential number ranging from 1 to MAX_SAMPLER_NUMBER.
153 // Same applies to other texture as the following list summarizes:
154 // diffuse: texture_diffuseN
155 // specular: texture_specularN
156 // normal: texture_normalN
157
158 // 1. diffuse maps
159 std::vector<Texture> diffuseMaps = LoadMaterialTextures(material, aiTextureType_DIFFUSE, "texture_diffuse");
160 textures.insert(textures.end(), diffuseMaps.begin(), diffuseMaps.end());
161 // 2. specular maps
162 std::vector<Texture> specularMaps = LoadMaterialTextures(material, aiTextureType_SPECULAR, "texture_specular");
163 textures.insert(textures.end(), specularMaps.begin(), specularMaps.end());
164 // 3. normal maps
165 std::vector<Texture> normalMaps = LoadMaterialTextures(material, aiTextureType_HEIGHT, "texture_normal");
166 textures.insert(textures.end(), normalMaps.begin(), normalMaps.end());
167 // 4. height maps
168 std::vector<Texture> heightMaps = LoadMaterialTextures(material, aiTextureType_AMBIENT, "texture_height");
169 textures.insert(textures.end(), heightMaps.begin(), heightMaps.end());
170
171 // return a mesh object created from the extracted mesh data
172 return Mesh(vertices, indices, textures);
173 }

175 std::vector<Texture> LoadMaterialTextures(aiMaterial *mat, aiTextureType type, std::string typeName)
176 {
177 std::vector<Texture> textures;
178 for (unsigned int i = 0; i < mat->GetTextureCount(type); i++)
179 {
180 aiString str;
181 mat->GetTexture(type, i, &str);
182 // check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
183 bool skip = false;
184 for (unsigned int j = 0; j < texturesLoaded.size(); j++)
185 {
186 if (std::strcmp(texturesLoaded[j].path.data(), str.C_Str()) == 0)
187 {
188 textures.push_back(texturesLoaded[j]);
189 skip = true; // a texture with the same filepath has already been loaded, continue to next one. (optimization)
190 break;
191 }
192 }
193 if (!skip)
194 { // if texture hasn't been loaded already, load it
195 Texture texture;
196 texture.Load(directory + str.C_Str());
197 texture.type = typeName;
198 textures.push_back(texture);
199 texturesLoaded.push_back(texture); // store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
200 }
201 }
202 return textures;
203 }
204 };
